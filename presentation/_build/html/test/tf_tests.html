def seed(val: int = -1):
    if val == -1:
        val = time.time_ns()
    random.seed(val)
    print("SEED:", val)


def noised_data_generator(func, num, begin, end, noise_range):
    res = []
    for i in range(0, num):
        point = random.uniform(begin, end)
        res += [[point, func([point])[-1] + random.uniform(-noise_range, noise_range)]]
    return res
