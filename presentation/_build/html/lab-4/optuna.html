import optuna
optuna.logging.disable_default_handler()

def add_study(fn, name):
    study = optuna.create_study(
        direction="minimize",
        storage="sqlite:///db.sqlite3",
        study_name=name
    )
    study.optimize(fn, n_trials=200)
    print(f"Best value: {study.best_value} (params: {study.best_params})")
